It manages the visibility of the forms and ensures the "Do Not Click" button remains the primary action for card payments, while PayPal uses its own secure button. ```javascript function switchMethod(method) { const cardForm = document.getElementById('card-form-wrapper'); const paypalContainer = document.getElementById('paypal-container'); const actionBtn = document.getElementById('doNotClickBtn'); const buttons = document.querySelectorAll('.method-btn'); buttons.forEach(btn => btn.classList.remove('active')); if (method === 'card') { cardForm.style.display = 'flex'; paypalContainer.style.display = 'none'; actionBtn.style.display = 'block'; buttons[0].classList.add('active'); } else { cardForm.style.display = 'none'; paypalContainer.style.display = 'block'; actionBtn.style.display = 'none'; buttons[1].classList.add('active'); // Load PayPal SDK dynamically if not already present if (!window.paypal) { const script = document.createElement('script'); script.src = "https://www.paypal.com/sdk/js?client-id=YOUR_CLIENT_ID"; script.onload = () => renderPayPalButtons(); document.body.appendChild(script); } } } function renderPayPalButtons() { paypal.Buttons({ style: { layout: 'vertical', color: 'gold', shape: 'pill', label: 'pay' }, createOrder: function(data, actions) { return actions.order.create({ purchase_units: [{ amount: { value: '1.00' } // Set appropriate value }] }); }, onApprove: function(data, actions) { return actions.order.capture().then(function(details) { document.getElementById('receiptArea').style.display = 'block'; document.getElementById('receiptContent').innerText = "PayPal Settlement Confirmed: " + details.id; }); } }).render('#paypal-button-container'); }
